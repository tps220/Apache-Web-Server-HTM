39 static APR_INLINE void FileTimeToAprTime(apr_time_t *result, FILETIME *input)
40 {
41  /* Convert FILETIME one 64 bit number so we can work with it. */
42  *result = input->dwHighDateTime;
43  *result = (*result) << 32;
44  *result |= input->dwLowDateTime;
45  *result /= 10; /* Convert from 100 nano-sec periods to micro-seconds. */
46  *result -= APR_DELTA_EPOCH_IN_USEC; /* Convert from Windows epoch to Unix epoch */
47  return;
48 }
